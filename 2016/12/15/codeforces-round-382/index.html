735A、Ostap and Grasshopper

思路：

　　从G往T跳，只要不经过#并且能到达就YES，否则NO。

code

#include <bits/stdc++.h>

#define rep(i, j, k) for(int i = (int) j; i < (int) k; ++i)
#define sz(x) ((int) (x).size())
#define ll long long
#define mp make_pair
#define pii pair<int, int >
#define fi first
#define se second
#define pb push_back
#define inf 0x3f3f3f3f
#define INF 0x3f3f3f3f3f3f3f
#define zero(x) memset((x), (0), sizeof (x))
#define zerox(x, y) memset((x), (y), sizeof (x))

using namespace std;

int main()
{
#ifdef PIT
//freopen("", "r", stdin);
//freopen(".out", "w", stdout);
#endif // PIT
    ios_base::sync_with_stdio(false); cin.tie(0);

    int n, k;
    string s;
    cin >> n >> k;
    cin >> s;
    int g, t;
    rep(i, 0, n) {
        if(s[i] == 'G') g = i;
        if(s[i] == 'T') t = i;
    }
    if(t < g) k = -k;
    bool f = 0;
    while(1) {
        if(g >= n || g < 0) break;
        g += k;
        if(s[g] == '#') break;
        if(g == t) {f = 1; break; }
    }
    if(f) cout << "YES";
    else cout << "NO";
    return 0;
}

735B、Urbanization

解法：

　　数组从大到小排序，交换n1, n2使得n1 < n2，先选前x个，再选前y个（证明略）。计算结果。

code

#include <bits/stdc++.h>

#define rep(i, j, k) for(int i = (int) j; i < (int) k; ++i)
#define sz(x) ((int) (x).size())
#define ll long long
#define mp make_pair
#define pii pair<int, int >
#define fi first
#define se second
#define pb push_back
#define inf 0x3f3f3f3f
#define INF 0x3f3f3f3f3f3f3f
#define zero(x) memset((x), (0), sizeof (x))
#define zerox(x, y) memset((x), (y), sizeof (x))

using namespace std;
const int N = 100100;

int a[N];
int n, n1, n2;

int main()
{
#ifdef PIT
//freopen("", "r", stdin);
//freopen(".out", "w", stdout);
#endif // PIT
    ios_base::sync_with_stdio(false); cin.tie(0);

    cin >> n >> n1 >> n2;
    rep(i, 0, n) cin >> a[i];
    sort(a, a+n);
    reverse(a, a+n);
    ll s1 = 0, s2 = 0;
    if(n1 > n2) swap(n1, n2);
    rep(i, 0, n1) s1 += a[i];
    rep(i, 0, n2) s2 += a[i+n1];
    if(n1 == 0) printf("%.8f\n", 1.* s2/n2);
    else printf("%.8f\n", 1.*s1/n1 + 1.*s2/n2);
    return 0;
}

735C、Tennis Championship

解法：

　　假设胜者一直是1，当其打了k场时与淘汰（不一定都与其比赛，可能和对手比赛被对手淘汰）至少为f[k]个人，那么下一个人能和其比赛的至少要打k-1场，那么这个人淘汰至少f[k-1]。那么如果1选手赢了比赛，其打了k+1场，淘汰f[k] + f[k-1]个人。至此，可以看出是fibonacci数列。

code

#include <bits/stdc++.h>

#define rep(i, j, k) for(int i = (int) j; i < (int) k; ++i)
#define sz(x) ((int) (x).size())
#define ll long long
#define mp make_pair
#define pii pair<int, int >
#define fi first
#define se second
#define pb push_back
#define inf 0x3f3f3f3f
#define INF 0x3f3f3f3f3f3f3f
#define zero(x) memset((x), (0), sizeof (x))
#define zerox(x, y) memset((x), (y), sizeof (x))

using namespace std;

int main()
{
#ifdef PIT
//freopen("", "r", stdin);
//freopen(".out", "w", stdout);
#endif // PIT
    ios_base::sync_with_stdio(false); cin.tie(0);

    ll n, a, b;
    cin >> n;
    int t = 0;
    for(a = 1, b = 2; b <= n; ++t) b = a + b, a = b - a;
    cout << t << endl;

    return 0;
}

735D、Taxes

解法：

　　哥德巴赫猜想，偶数可化成2个质数相加（数据范围内）。奇数可化为一个奇数+一个偶数。枚举特例就行（数字2）。

code

#include <bits/stdc++.h>

#define rep(i, j, k) for(int i = (int) j; i < (int) k; ++i)
#define sz(x) ((int) (x).size())
#define ll long long
#define mp make_pair
#define pii pair<int, int >
#define fi first
#define se second
#define pb push_back
#define inf 0x3f3f3f3f
#define INF 0x3f3f3f3f3f3f3f
#define zero(x) memset((x), (0), sizeof (x))
#define zerox(x, y) memset((x), (y), sizeof (x))

using namespace std;

bool isprime(int x) {
    for(int i = 2; i * i <= x; ++i)
        if(x % i == 0) return 0;
    return 1;
}

int main()
{
#ifdef PIT
//freopen("", "r", stdin);
//freopen(".out", "w", stdout);
#endif // PIT
    ios_base::sync_with_stdio(false); cin.tie(0);

    int n;
    cin >> n;
    if(n&1) {
        if(isprime(n)) cout << 1 << endl;
        else if(isprime(n-2)) cout << 2 << endl;
        else cout << 3 << endl;
    }
    else if(n == 2) cout << 1 << endl;
    else cout << 2 << endl;
    return 0;
}
