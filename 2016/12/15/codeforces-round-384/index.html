743A、Vladik and flights

思路：

　　s[a] == s[b] ? "0" : "1"

code

#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios_base::sync_with_stdio(false); cin.tie(0);
    int n, a, b;
    string s;
    cin >> n >> a >> b;
    cin >> s;
    --a, --b;
    if(s[a] == s[b]) cout << 0;
    else cout << 1;
    return 0;
}

743B、Chloe and the sequence

解法：

　　当在中间位置时，直接输出n；否则，左右相等，如果在右边，对应到左边位置，n--，直到找到中间位置输出。

code

#include <bits/stdc++.h>

using namespace std;
const int N = 52;

ll b[N];

void init() {
    b[0] = 1;
    for(int i = 1; i < N; ++i) b[i] = (b[i-1]<<1);
    return ;
}

int main()
{
    ios_base::sync_with_stdio(false); cin.tie(0);
    init();
    int n;
    ll k;
    cin >> n >> k;
    while(k && n) {
        if(k == b[n-1]) {
            cout << n;
            break;
        }
        if(k > b[n-1]) k -= b[n-1];
        --n;
    }
    return 0;
}

743C、Vladik and fractions

解法：

　　x = n, y = n+1, z = (n+1)*n，并且x, y, z不相等时成立（n=1的情况）。

code

#include <bits/stdc++.h>
using namespace std;

ll x, y, z, n;

int main()
{
    ios_base::sync_with_stdio(false); cin.tie(0);
    cin >> n;
    if(n == 1) cout << -1 << endl;
    else{
        x = n; y = n + 1;
        z = x * y;
        cout << x << " " << y << " " << z << endl;
    }
    return 0;
}

743D、Chloe and pleasant prizes

解法：

　　原先做法：先求dfs序，求出去每个节点获得最大的pleasant，对pleasant排序，取dfs序不想交的2个最大的值求和。然后TLE了。
　　正确做法：树上dp，dp[u]表示u节点及其子树中的pleasant最大值。

code

#include <bits/stdc++.h>

#define rep(i, j, k) for(int i = (int) j; i < (int) k; ++i)
#define sz(x) ((int) (x).size())
#define ll long long
#define mp make_pair
#define pii pair<int, int >
#define fi first
#define se second
#define pb push_back
#define inf 0x3f3f3f3f
#define INF 0x3f3f3f3f3f3f3f
#define zero(x) memset((x), (0), sizeof (x))
#define zerox(x, y) memset((x), (y), sizeof (x))

using namespace std;
const int N = 200020;

vector<int > E[N];
ll ple[N];
ll dp[N];
int n;
ll ans;

void dfs(int u, int p) {
    for(int v : E[u]) if(v != p) {
        dfs(v, u);
        ple[u] += ple[v];
        //此时dp[u]记录的是已经遍历的子树的max值
        if(dp[u] != -INF) ans = max(ans, dp[u] + dp[v]); // 取子树内2棵不相交子树的max值和
        dp[u] = max(dp[u], dp[v]);
    }
    //节点u所有子树的max值
    dp[u] = max(dp[u], ple[u]);
}

int main()
{
//freopen("", "r", stdin);
//freopen(".out", "w", stdout);

    ios_base::sync_with_stdio(false); cin.tie(0);
    cin >> n;
    rep(i, 1, n+1) cin >> ple[i], dp[i] = -INF;
    int x, y;
    rep(i, 1, n) {
        cin >> x >> y;
        E[x].pb(y); E[y].pb(x);
    }
    ans = -INF;
    dfs(1, 0);
    if(ans == -INF) cout << "Impossible";
    else cout << ans;
    return 0;
}
